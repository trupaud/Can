#include <avr/io.h>
#include <avr/interrupt.h>


#define TIEMPO_A 2   
#define TIEMPO_B 750 


void configurarTimer1() {
  
    TCCR1B |= (1 << WGM12);


    TCCR1B |= (1 << CS11) | (1 << CS10);

    OCR1A = (TIEMPO_A * 20000) / 64 - 1;
    OCR1B = (TIEMPO_B * 20000) / 64 - 1;
    TIMSK1 |= (1 << OCIE1A) | (1 << OCIE1B);
}


void inicializar() {
    configurarTimer1();
    sei();
}

int main() {
    inicializar();

    while (1) {
    }
    return 0;
}
